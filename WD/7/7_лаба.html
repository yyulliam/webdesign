Лабораторная работа №7 - Вариант 9

1. isNegative(num) - Проверка отрицательного числа

function isNegative(num) { return num < 0; }

2. trimSpaces(str) - Удаление пробелов

function trimSpaces(str) { return str.trim(); }

3. andNot(a, b) - Логическое "И НЕ"

function andNot(a, b) { return !(a && b); }

4. isValueEqual(obj, key, value) - Проверка свойства объекта

function isValueEqual(obj, key, value) { return obj[key] === value; }

5. contains(array, element) - Проверка элемента в массиве

function contains(array, element) { return array.includes(element); }

6. logEven(array) - Вывод четных чисел

function logEven(array) { array.forEach(num => { if (num % 2 === 0) console.log(num); }); }

7. squareNumbers(array) - Квадраты чисел

function squareNumbers(array) { return array.map(num => num * num); }
